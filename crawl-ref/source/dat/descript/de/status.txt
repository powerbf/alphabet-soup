Water status

{{
    local feat = view.feature_at(0, 0):gsub("_", " ")
    if you.flying() then
        if feat == "deep water" then
            return "Ihr fliegt über tiefes Wasser.
        elseif feat == "shallow water" then
            return "Ihr fliegt über seichtes Wasser.
        else
            return "Ihr fliegt über Wasser.
        end
    else
        local verb = you.race() == "Merfolk" and "schwimmt in" or "steht in"
        if feat == "deep water" then
            return "Ihr " .. verb .. " tiefem Wasser.
        elseif feat == "shallow water" then
            return "Ihr " .. verb .. " seichtem Wasser.
        else
            return "Ihr " .. verb .. " Wasser.
        end
    end
}}
